src/

│

├── pages/                   # Route-based components

│   └── rootReducer.ts
│
├── services/                # API calls and socket connections

│   ├── formatDate.ts

│   │   └── profileSchema.ts
│   └── constants.ts         # Common constants (API_URL, roles, etc.)
│
├── types/                   # Global TypeScript types/interfaces
│   ├── user.ts
│   ├── message.ts
│   └── chat.ts
│
├── hooks/                   # Custom reusable React hooks
│   ├── useAuth.ts
│   ├── useChat.ts
│   └── useSocket.ts
│
├── routes/                  # Route definitions and protected routes
│   ├── AppRouter.tsx
│   └── ProtectedRoute.tsx
│
├── App.tsx                  # Root app component
├── main.tsx                 # ReactDOM.render (entry point)
